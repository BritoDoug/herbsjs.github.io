ENTITIES

const { entity, field } = require('gotu')
const Feature = 
        entity('Feature', {
            name: field(String),
            hasAccess: field(Boolean)
        })
const Plan = 
    entity('Plan', {
        name: field(String),
        monthlyCost: field(Number)
    })
const User = 
    entity('User', {
        name: field(String),
        lastAccess: field(Date),
        accessCount: field(Number),
        features: field([Feature]),
        plan: field(Plan),
    })
const user = new User()

    new Feature(),

]
